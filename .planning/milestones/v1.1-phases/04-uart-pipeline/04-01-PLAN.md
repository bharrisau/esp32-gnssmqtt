use esp_idf_svc::hal::peripheral::Peripheral;

use esp_idf_svc::hal::units::Hertz;
use std::sync::Arc;

let uart = UartDriver::new(
    uart,
    tx_pin,
    rx_pin,
    Option::<AnyIOPin>::None,
    Option::<AnyIOPin>::None,
    &Config::new()
        .baudrate(Hertz(115_200))
        .rx_fifo_size(crate::config::UART_RX_BUF_SIZE as usize),
)?;

// Arc sharing pattern (UartDriver is Send; read/write take &self):
let uart = Arc::new(uart);
let uart_rx = Arc::clone(&uart);
// Both threads can call uart_rx.read() and uart.write() concurrently via &self

// NON_BLOCK polling loop (avoids FreeRTOS watchdog):
let mut buf = [0u8; 256];
loop {
    match uart_rx.read(&mut buf, NON_BLOCK) {
        Ok(n) if n > 0 => { /* process */ }
        _ => std::thread::sleep(std::time::Duration::from_millis(10)),
    }
}

// Thread spawn pattern (universal in codebase):
std::thread::Builder::new()
    .stack_size(8192)
    .spawn(move || { /* body */ })
    .expect("thread name spawn failed");

**Public function signature (LOCKED — do not change):**
```rust
pub fn spawn_gnss(
    uart: impl Peripheral<P = impl Uart> + 'static,
    tx_pin: impl Peripheral<P = impl esp_idf_svc::hal::gpio::OutputPin> + 'static,
    rx_pin: impl Peripheral<P = impl esp_idf_svc::hal::gpio::InputPin> + 'static,
) -> anyhow::Result<(Sender<String>, Receiver<(String, String)>)>
```

**Body of spawn_gnss:**
1. Create UartDriver using the exact initialization pattern from uart_bridge.rs (see interfaces block). GPIO16=TX, GPIO17=RX, 115200 baud, rx_fifo_size=crate::config::UART_RX_BUF_SIZE.
2. Wrap in `Arc::new(uart)` immediately — UartDriver is not Clone; Arc allows both threads to share via &self.
3. Create two unbounded channels:
   - `let (nmea_tx, nmea_rx) = mpsc::channel::<(String, String)>();`
   - `let (cmd_tx, cmd_rx) = mpsc::channel::<String>();`
4. Spawn RX thread (see below). Move `Arc::clone(&uart)` and `nmea_tx` into it.
5. Spawn TX thread (see below). Move the original `uart` Arc and `cmd_rx` into it.
6. Return `Ok((cmd_tx, nmea_rx))`

**RX thread body** — NON_BLOCK polling + sentence assembly:
- Stack size: 8192
- Variables: `let mut read_buf = [0u8; 256];` (read buffer), `let mut line_buf = [0u8; 512];` (accumulator, 512 bytes — covers UM980 proprietary sentences), `let mut line_len: usize = 0;`
- Loop: `match uart_rx.read(&mut read_buf, NON_BLOCK)` → on Ok(n) if n > 0, iterate over `&read_buf[..n]`:
  - If `byte == b'\n'`: process the accumulated line:
    - Strip trailing `\r`: `let end = if line_len > 0 && line_buf[line_len-1] == b'\r' { line_len-1 } else { line_len };`
    - Mirror to stdout: `let _ = std::io::stdout().write_all(&line_buf[..end]); let _ = std::io::stdout().write_all(b"\n");`
    - If `line_buf[..end].first() == Some(&b'$')` AND end > 1:
      - Convert to str: `if let Ok(s) = std::str::from_utf8(&line_buf[..end])`
      - Extract type: `let sentence_type = s[1..].split(',').next().unwrap_or("UNKNOWN").to_string();`
      - Send: `let _ = nmea_tx.send((sentence_type, s.to_string()));`
    - Else if end > 0: `log::warn!("GNSS: non-NMEA line dropped: {:?}", std::str::from_utf8(&line_buf[..end]));`
    - (Empty lines after stripping: end == 0, skip silently — no log)
    - Reset: `line_len = 0;`
  - Else if `line_len < line_buf.len()`: `line_buf[line_len] = byte; line_len += 1;`
  - Else (buffer full): `log::warn!("GNSS: RX line buffer overflow, discarding {} bytes", line_len); line_len = 0;`
- On read returning 0 or error: `std::thread::sleep(std::time::Duration::from_millis(10));`
- Thread name for expect: "gnss rx spawn failed"

**TX thread body** — blocking mpsc drain → UART write:
- Stack size: 8192
- Loop: `for line in &cmd_rx` (blocking recv — thread sleeps until a command arrives; correct for infrequent TX)
  - `let _ = uart_tx.write(line.as_bytes());`
  - `let _ = uart_tx.write(b"\r\n");` (UM980 requires CRLF)
- After loop exits (all Senders dropped): `log::error!("GNSS TX channel closed — TX thread exiting");`
- Thread name for expect: "gnss tx spawn failed"

**Conventions:**
- All logging uses `log::info!`, `log::warn!`, `log::error!` — never println!
- Only `std::io::stdout().write_all()` for raw byte mirroring (not log)
- UART errors in RX loop: log at warn, continue (never panic)
- Do NOT add `mod gnss;` to main.rs — that is Plan 02's job
