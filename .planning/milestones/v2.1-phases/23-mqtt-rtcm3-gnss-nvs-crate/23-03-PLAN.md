Epoch grouping pattern (from RESEARCH.md Pattern 3):
```rust
struct EpochBuffer {
    epoch_key: u32,
    observations: Vec<Obs>,
}
impl EpochBuffer {
    fn push(&mut self, epoch_ms: u32, obs: Vec<Obs>) -> Option<EpochGroup> {
        if self.epoch_key != 0 && epoch_ms != self.epoch_key {
            let group = self.flush(); // take accumulated
            self.epoch_key = epoch_ms;
            self.observations = obs;
            return Some(group);
        }
        self.epoch_key = epoch_ms;
        self.observations.extend(obs);
        None
    }
}
```
